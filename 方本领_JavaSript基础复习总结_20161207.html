2、字符串的方法

indexOf()是用方法：

如何使用indexof()来定位字符串某一个指定的字符首次出现的位置

 var str = "hello world";
使用indexof索引，得到的索引号是从0开始的，
   indexOf(d);
   console.log(str.indexOf("d"));这是我们索引的方式
有时候为了方便我们会声明一个函数：
function trace(bgm){
	console.log(bgm);
}
  trace(str.indexOf("d"));
             索引号:11
后面的可以用trace();代替console.log();去检查元素。

substr():从起始索引号提取字符串中指定数目的字符。substr(start,length)

		function trace(bgm){
		console.log(bgm);
	}
		var str = "abcdef";
		trace(str.substr(0,4));
		结果:abcd

substring():提取字符串中两个指定的索引号之间的字符。substring(from,to)

function trace(bgm)
{console.log(bgm);
	}
var str = "abcdef";
trace(str.substring(2,6));   
  结果:cdef

slice():提取字符串的片断，并在新的字符串中返回被提取的部分。slice(start,end)

var str = "abcdef";
trace(str.slice(0,6));   
  结果:abcdef

split():把字符串分割为字符串数组。split("",num) 引号内部是要切割字符的部分，mun是切割成几部分

		var str = "abcdef";
		trace(str.split("c",3));
		结果："ab,def"  c自己本身被切除,后面的数字是根据切除节点的个数后需要留下数组内部元素的个数
如果2个c:
var str = "abcdecf";
 trace(str.split("c",3));
 结果："ab,de,f"
 trace(str.split("c",2));
 结果："ab,de"

charAt():可返回指定位置的字符。charAt(num)


var str="Hello world!"
document.write(str.charAt(1))
结果：e

concat():连接字符串，用于连接两个或多个字符串

提示：请注意，使用 " + " 运算符来进行字符串的连接运算通常会更简便一些。 

var str1="Hello "
var str2="world!"
document.write(str1.concat(str2));
结果:Hello world

match():可在字符串内检索指定的值，或找到一个或多个正则表达式的匹配。 该方法类似indexof()，但是它返回指定的值，而不是字符串的位置。 

var str="Hello world!"
document.write(str.match("world") + "
")
document.write(str.match("World") + "
")
document.write(str.match("worlld") + "
")
document.write(str.match("world!"))
结果
world
null
null
world!

var str="1 plus 2 equal 3"
document.write(str.match(/\d+/g))
结果：1、2、3
